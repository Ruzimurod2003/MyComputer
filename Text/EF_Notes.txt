	public class MY_DB: DbContext
    {
		//Bu class 'dotnet ef database update' buyrugidan keyin

	public class Student
    {
		//[Key] -- bazada 'not null primary key' degani
        [Key]
        public Guid Id { get; set; }
		
		//[Required]-- bazada 'not null' degani 
        [Required]
        public string Name { get; set; }

3. HomeController.cs da database dagi tablelarni o'zgartiramiz:

	public class HomeController : ControllerBase
    {
		//MY_DB clasidan namuna olib uni ishga tushiramiz
        private MY_DB db;
        public HomeController(MY_DB db)
        {
            this.db = db;
        }		
		
		
		//Sinov uchun metod
        public void SomeTestMethod(int id, Student student)
        {
			//barcha talabalar uchun(ToList)
            IEnumerable<Student> _students=db.Students.ToList();
			
			//Bitta talaba uchun
			Student _student=db.Students.Find(id);			
			
			//Bitta talabani qo'shish(Add)
			db.Students.Add(student);
            db.SaveChanges();	

			//Talaba ma'lumotlarini yangilaydi(Update)
			db.Students.Update(student);
            db.SaveChanges();
			
			//Talabani o'chirish(Remove)
			var student = db.Students.Find(id);
            db.Students.Remove(student);
            db.SaveChanges();
        }
	}
